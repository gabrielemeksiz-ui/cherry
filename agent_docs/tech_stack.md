## Deezer Pagination Pattern (for playlists > 100 tracks)
```javascript
// Fetch ALL tracks from a Deezer playlist — no size limit
async function getAllDeezerTracks(playlistId, deezerToken) {
  let tracks = [];
  let offset = 0;
  const limit = 100;
  let total = Infinity;

  while (tracks.length < total) {
    const response = await fetch(
      `https://api.deezer.com/playlist/${playlistId}/tracks?limit=${limit}&index=${offset}`,
      { headers: { Authorization: `Bearer ${deezerToken}` } }
    );
    const data = await response.json();
    total = data.total;
    tracks = [...tracks, ...data.data];
    offset += limit;
  }

  return tracks; // Complete list, no limit
}
```

## Spotify Rate Limit Pattern
```javascript
// Add 100ms delay between batch requests to avoid hitting Spotify rate limits
async function addTracksToSpotify(spotifyToken, playlistId, trackUris) {
  const BATCH_SIZE = 100;
  for (let i = 0; i < trackUris.length; i += BATCH_SIZE) {
    const batch = trackUris.slice(i, i + BATCH_SIZE);
    await fetch(`https://api.spotify.com/v1/playlists/${playlistId}/tracks`, {
      method: 'POST',
      headers: {
        Authorization: `Bearer ${spotifyToken}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ uris: batch }),
    });
    await new Promise(resolve => setTimeout(resolve, 100)); // Rate limit buffer
  }
}
```

```xml
<dict>
  <key>Playlists</key>
  <array>
    <dict>
      <key>Name</key><string>My Playlist</string>
      <key>Playlist Items</key>
      <array>
        <dict>
          <key>Track ID</key><integer>123</integer>
        </dict>
      </array>
    </dict>
  </array>
  <key>Tracks</key>
  <dict>
    <key>123</key>
    <dict>
      <key>Name</key><string>Song Title</string>
      <key>Artist</key><string>Artist Name</string>
      <key>Album</key><string>Album Name</string>
    </dict>
  </dict>
</dict>
```

## Supabase Database Schema
```sql
CREATE TABLE transfers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  playlist_name TEXT NOT NULL,
  source TEXT NOT NULL,         -- 'deezer' or 'apple_music'
  total_tracks INTEGER NOT NULL,
  transferred_tracks INTEGER NOT NULL,
  failed_tracks INTEGER NOT NULL DEFAULT 0,
  transferred_at TIMESTAMP DEFAULT NOW()
);
```
